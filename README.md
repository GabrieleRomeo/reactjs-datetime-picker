## Install

```bash
npm install --save reactjs-datetime-picker
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'reactjs-datetime-picker'
import 'reactjs-datetime-picker/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
